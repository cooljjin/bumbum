  // 마지막 액션 표시
  useEffect(() => {
    let timer: NodeJS.Timeout | undefined;
    
    if (history.past.length > 0) {
      const lastItem = history.past[history.past.length - 1];
      if (lastItem && lastItem.description) {
        setLastAction(lastItem.description);
        timer = setTimeout(() => setLastAction(''), 3000);
      }
    }
    
    return () => {
      if (timer) {
        clearTimeout(timer);
      }
    };
  }, [history.past]);
